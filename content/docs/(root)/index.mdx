## Installation

<CodeTabs>

<TabsContent value="cli">

```bash
npx shadcn@latest add @shadcn-map/map
```

</TabsContent>

</CodeTabs>

## Usage

```tsx showLineNumbers
import {
    Map,
    MapMarker,
    MapPopup,
    MapTileLayer,
    MapZoomControl,
} from "@/registry/new-york-v4/ui/map";
```

```tsx showLineNumbers
<Map center={[43.6532, -79.3832]}>
    <MapTileLayer />
    <MapZoomControl />
    <MapMarker position={[43.6532, -79.3832]}>
        <MapPopup>A map component for shadcn/ui.</MapPopup>
    </MapMarker>
</Map>
```
